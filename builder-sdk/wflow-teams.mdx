```python
from agno import Team

# Create individual agents first
agent1 = Agent(id="researcher", model=OpenAIChat(id="gpt-4"), db=db)
agent2 = Agent(id="writer", model=OpenAIChat(id="gpt-4"), db=db)

agent1.save()
agent2.save()

# Create a team
team = Team(
    id="content-team",
    name="Content Creation Team",
    members=[agent1, agent2],
    db=db
)

# Save team (creates entries in agno_component_links)
team.save()
```
